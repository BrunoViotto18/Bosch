#include <stdio.h>
#include <stdlib.h>
#include <locale.h>
#include <stdbool.h>
struct cadastro{
    char nome[500];
    char email[500];
    int dia_nasc;
    int mes_nasc;
    int ano_nasc;
    long int cpf;
    int sexo;
};

void imprimir (struct cadastro c){
    printf ("Nome: %s\n", c.nome);
    printf ("email: %s\n", c.email);
    printf ("Nasc: %d/%d/%d \n", c.dia_nasc, c.mes_nasc, c.ano_nasc);
    printf ("CPF: %ld \n", c.cpf);
    printf ("sexo: %d \n", c.sexo);
}
int main()
{
    struct cadastro c;
    printf ("Digite o nome");
    scanf ("%s", &c.nome);

    printf ("Digite o email");
    scanf ("\n%s", &c.email);

    printf ("Digite o dia de nasc");
    scanf ("\n%d", &c.dia_nasc);

    printf ("Digite o mes de nasc");
    scanf ("%d", &c.mes_nasc);

    printf ("Digite o ano de nasc");
    scanf ("%d", &c.ano_nasc);

    printf ("Digite o cpf");
    scanf ("\n%ld", &c.cpf);

    printf ("Digite o sexo");
    scanf ("\n%d", &c.sexo);

    imprimir (c);

    return 0;
}
